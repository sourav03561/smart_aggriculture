-- Query for average temperature sensor data for each land within a specified time period
SELECT 
    sd.land_id, 
    AVG(sd.value) AS avg_temperature
FROM 
    Sensor_Data sd
INNER JOIN 
    Sensors s ON sd.sensor_id = s.sensor_id
INNER JOIN 
    Sensor_Type st ON s.sensor_typeid = st.sensor_typeid
WHERE 
    st.typename = 'Temperature' AND
    sd.date BETWEEN 'Start_Date' AND 'End_Date'
GROUP BY 
    sd.land_id;
